    let currentIndex = 0;
    let showChunks = false;
    let showMeaning = false;

    const progressChip = document.getElementById("progressChip");
    const qBadge = document.getElementById("qBadge");
    const qTitle = document.getElementById("qTitle");
    const levelChip = document.getElementById("levelChip");
    const passage = document.getElementById("passage");
    const chunkList = document.getElementById("chunkList");
    const miniInfo = document.getElementById("miniInfo");
    const prevBtn = document.getElementById("prevBtn");
    const nextBtn = document.getElementById("nextBtn");
    const toggleChunksBtn = document.getElementById("toggleChunksBtn");
    const toggleMeaningBtn = document.getElementById("toggleMeaningBtn");

    function renderQuestion(){
      const q = questions[currentIndex];

      progressChip.textContent = `${currentIndex + 1} / ${questions.length}`;
      qBadge.textContent = `Q${q.no}`;
      qTitle.textContent = q.title;
      levelChip.textContent = q.level;
      passage.textContent = q.passage;
      miniInfo.textContent = q.info;

      chunkList.innerHTML = "";

      q.chunks.forEach(item => {
        const card = document.createElement("div");
        card.className = "chunk-card" + (showChunks ? "" : " chunk-hidden");

        const en = document.createElement("div");
        en.className = "chunk-en";
        en.textContent = item.en;

        const ko = document.createElement("div");
        ko.className = "chunk-ko" + (showMeaning ? "" : " chunk-hidden");
        ko.textContent = "= " + item.ko;

        card.appendChild(en);
        card.appendChild(ko);
        chunkList.appendChild(card);
      });

      prevBtn.disabled = currentIndex === 0;
      nextBtn.disabled = currentIndex === questions.length - 1;

      toggleChunksBtn.classList.toggle("active", showChunks);
      toggleMeaningBtn.classList.toggle("active", showMeaning);

      toggleChunksBtn.textContent = showChunks ? "chunk 숨기기" : "chunk 보기";
      toggleMeaningBtn.textContent = showMeaning ? "뜻 숨기기" : "뜻 보기";
    }

    toggleChunksBtn.addEventListener("click", () => {
      showChunks = !showChunks;
      renderQuestion();
    });

    toggleMeaningBtn.addEventListener("click", () => {
      showMeaning = !showMeaning;
      renderQuestion();
    });

    prevBtn.addEventListener("click", () => {
      if (currentIndex > 0){
        currentIndex--;
        renderQuestion();
      }
    });

    nextBtn.addEventListener("click", () => {
      if (currentIndex < questions.length - 1){
        currentIndex++;
        renderQuestion();
      }
    });

    document.addEventListener("keydown", (e) => {
      if (e.key === "ArrowLeft" && currentIndex > 0){
        currentIndex--;
        renderQuestion();
      }
      if (e.key === "ArrowRight" && currentIndex < questions.length - 1){
        currentIndex++;
        renderQuestion();
      }
      if (e.key === " "){
        e.preventDefault();
        showChunks = !showChunks;
        renderQuestion();
      }
    });

    renderQuestion();
